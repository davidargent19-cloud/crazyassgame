<script>

const groupContainer = document.getElementById("groups");

// Create 10 input blocks

for (let i = 0; i < 10; i++) {

    const div = document.createElement("div");

    div.className = "group";

    div.innerHTML = `

        <input placeholder="Group Name">

        <input type="color">

        <textarea rows="2" placeholder="Enter 10 words, comma-separated"></textarea>

    `;

    groupContainer.appendChild(div);

}

function generateLink() {

    const groupEls = document.querySelectorAll(".group");

    let groups = [];

    for (let g of groupEls) {

        const name = g.children[0].value.trim();

        const color = g.children[1].value;

        const words = g.children[2].value.split(",").map(w => w.trim()).filter(Boolean);

        if (!name || words.length !== 10) {

            alert("Each group must have a name and exactly 10 words.");

            return;

        }

        groups.push({ name, color, words });

    }

    const encoded = btoa(JSON.stringify(groups));

    const link = location.origin + location.pathname + "#puzzle=" + encoded;

    document.getElementById("linkOutput").innerHTML =

        <strong>Share this link:</strong><br><a href="${link}" target="_blank">${link}</a>;

}

// =====================

// GAME LOGIC

// =====================

const hash = location.hash;

if (hash.startsWith("#puzzle=")) {

    document.getElementById("builder").style.display = "none";

    document.getElementById("game").style.display = "block";

    const groups = JSON.parse(atob(hash.replace("#puzzle=", "")));

    startGame(groups);

}

let tiles = [];

let selected = [];

const grid = document.getElementById("grid");

const message = document.getElementById("message");

function startGame(groups) {

    groups.forEach(g => {

        g.words.forEach(word => {

            tiles.push({

                label: word,

                group: g.name,

                color: g.color,

                size: 1

            });

        });

    });

    tiles.sort(() => Math.random() - 0.5);

    render();

}

function render() {

    grid.innerHTML = "";

    tiles.forEach((t, i) => {

        const div = document.createElement("div");

        div.className = "tile";

        div.style.background = t.color;

        if (selected.includes(i)) div.classList.add("selected");

        div.innerHTML = t.label;

        div.onclick = () => toggle(i);

        grid.appendChild(div);

    });

}

function toggle(i) {

    selected.includes(i)

        ? selected = selected.filter(x => x !== i)

        : selected.push(i);

    render();

    check();

}

function check() {

    const total = selected.reduce((s,i)=>s+tiles[i].size,0);

    if (total !== 10) return;

    const group = tiles[selected[0]].group;

    if (!selected.every(i => tiles[i].group === group)) {

        selected = [];

        render();

        return;

    }

    const els = [...grid.children];

    selected.forEach(i => els[i].classList.add("collapsing"));

    setTimeout(() => {

        const color = tiles[selected[0]].color;

        const words = selected.map(i=>tiles[i].label).join("<br>");

        selected.sort((a,b)=>b-a).forEach(i=>tiles.splice(i,1));

        tiles.push({

            label:<div class="group-label">${group}</div>${words},

            group,

            color,

            size:10

        });

        selected = [];

        render();

        if (tiles.length === 1) message.textContent = "🎉 Puzzle Complete!";

    }, 300);

}
